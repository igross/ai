\begin{frame}
\frametitle{Motivation}
\begin{itemize}
\item Aggregate entrepreneurship is often higher in lower-income economies.
\item But composition matters: necessity entrepreneurship is much more prevalent where formal jobs and insurance are limited.
\item Our goal is to separate \textit{opportunity} vs \textit{necessity} entrepreneurship using optimal behavior, not self-reported motives.
\item Central policy question: how unemployment insurance (UI) affects occupational sorting and misallocation.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Definitions and Identification}
\begin{itemize}
\item Existing empirical proxy: previous employment status (employed = opportunity, unemployed = necessity).
\item Limitation: this can misclassify agents with strategic entry timing.
\item In our model, necessity entrepreneurship is defined by a \textbf{counterfactual occupational choice}:
\begin{itemize}
\item Necessity entrepreneur: chooses entrepreneurship only when unemployed.
\item Opportunity entrepreneur: chooses entrepreneurship regardless of employment status.
\end{itemize}
\item This maps naturally to cutoff rules by education and labor-market status.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Research Questions}
\begin{itemize}
\item How much of entrepreneurship is necessity-driven in equilibrium?
\item Does stronger UI reduce necessity entry and improve allocation?
\item Are effects concentrated by education group?
\item Do financing and matching frictions amplify these effects?
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Economic Environment}
\begin{itemize}
\item Infinite-horizon economy with occupational choice and search frictions.
\item State variables: $(h,a,x,z,i_w)$.
\item Education groups: $h \in \{P,S,T\}$.
\item Labor status: unemployed, employed, entrepreneur.
\item Occupational choice each period: worker vs entrepreneur.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Firm Problem}
\[
\max_{n}\; xk^{\alpha}(zn)^{\gamma} - zw(1+\psi\tau)n - rk - \kappa\frac{\rho zn}{q(\theta)}
\]
\begin{itemize}
\item Output: span-of-control technology with managerial talent $x$.
\item Hiring/search costs rise when vacancy filling is harder ($q(\theta)$ lower).
\item Main implication: weaker labor markets raise effective labor costs and shrink efficient firm scale.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Household Problem and Worker Income Risk}
\[
\mathbf{V}(a,h,z,x,i_w)=\max_{\{a',c,\mathcal{I}_e\}}\{\cdots\}
\]
\[
\tilde w_n=(1-\rho)\{zw(1-(1-\psi)\tau)\}+\rho b,
\quad
\tilde w_u=p(\theta)\{zw(1-(1-\psi)\tau)\}+(1-p(\theta))b
\]
\begin{itemize}
\item Employed workers face separation risk $\rho$.
\item Unemployed/ex-entrepreneurs face job-finding risk $p(\theta)$.
\item Different continuation risks generate necessity vs opportunity margins.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Cutoff Characterization}
\begin{itemize}
\item Let $\hat h_{i_w=1}$ and $\hat h_{i_w\neq 1}$ be occupation cutoffs.
\item Necessity region: $\hat h_{i_w\neq 1}<h<\hat h_{i_w=1}$.
\item Opportunity region: $h\geq\hat h_{i_w=1}$.
\item Six cutoffs in total: two per education group $(P,S,T)$.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Main Takeaways}
\begin{itemize}
\item Necessity entrepreneurship is an equilibrium response to unemployment risk.
\item UI reduces necessity entry and can improve allocation by discouraging low-productivity self-employment.
\item Education heterogeneity matters for both entry and firm size responses.
\item Policy design should account for both insurance benefits and occupational-sorting distortions.
\end{itemize}
\end{frame}
