\begin{frame}
\frametitle{Issue 1 (High): Main vs Appendix Technology Mismatch}
\begin{itemize}
\item Main text uses production proportional to $xk^{\alpha}(zn)^{\gamma}$:
\begin{itemize}
\item \texttt{maxP} (lyx:1078), \texttt{profitj} (lyx:1175)
\end{itemize}
\item Appendix uses $x^{1-\alpha-\gamma}k^{\alpha}(zn)^{\gamma}$:
\begin{itemize}
\item \texttt{eq:xstar\_u}, \texttt{eq:xstar\_n} (lyx:8866--8867)
\item \texttt{eq:profit\_emtrepreneur} (lyx:9111)
\end{itemize}
\item This can shift talent returns, cutoffs, and entrepreneurship shares.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Issue 2 (High): Entrepreneur Continuation State Inconsistency}
\begin{itemize}
\item State labels define entrepreneur as a distinct state:
\begin{itemize}
\item main text (lyx:1373--1382), appendix (lyx:8786--8795)
\end{itemize}
\item But Bellman terms send entrepreneur choice to $i'_w=0$:
\begin{itemize}
\item main Bellman (lyx:1454--1457)
\item appendix Bellman \texttt{eq:V\_overall} block (lyx:8907--8919)
\end{itemize}
\item Need one clear transition convention for entrepreneurship status.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Issue 3 (Medium): Formal Typo in Appendix Bellman}
\begin{itemize}
\item In \texttt{eq:V\_iw0\_xh}, continuation state is written as $i'_w=w$.
\item Location: lyx:9094.
\item This is almost surely a typo, but should be fixed for formal validity.
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Issue 4 (High): Cutoff in Flow Terms vs Dynamic Values}
\begin{itemize}
\item Appendix cutoff equations set $\tilde{w}=\pi(x^*)$:
\begin{itemize}
\item \texttt{eq:xstar\_u}, \texttt{eq:xstar\_n} (lyx:8866--8867)
\end{itemize}
\item Dynamic decision is defined by Bellman value comparison:
\begin{itemize}
\item \texttt{eq:V\_overall} block (lyx:8907--8922)
\end{itemize}
\item Need either:
\begin{itemize}
\item value-equality cutoffs, or
\item explicit assumptions that justify flow-equality shortcut.
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Issue 5-6 (Medium): Notation Consistency Gaps}
\begin{itemize}
\item Tax symbol inconsistency:
\begin{itemize}
\item uses $\tau_y$ in worker budgets (lyx:1259, 1299)
\item uses $\tau$ in expected worker income (lyx:1496--1500, 8885--8886)
\end{itemize}
\item Bellman continuation notation drops state components:
\begin{itemize}
\item LHS: $V(a,h,z,x,i_w)$
\item continuation written as $V(a',x',i'_w)$ at lyx:1454--1457
\end{itemize}
\item Likely fixable as notation cleanup, but important for replication clarity.
\end{itemize}
\end{frame}
